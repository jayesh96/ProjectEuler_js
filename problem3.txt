function max_prime_factor(n){
 
while(n%2 == 0){
	maxPrime = 2;
	n /= 2
}
 
for (var i = 3; i <= Math.sqrt(n); i += 2) {
        while (n % i == 0) {
            maxPrime = i;
            n = n / i;
        }
    }
 
 if (n > 2)
        maxPrime = n;
 
    return maxPrime;
 
}
